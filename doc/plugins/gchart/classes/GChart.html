encode(encoding, n, max)

Encode n as a string. n is normalized based on max. encoding can currently only be :extended.

[Source]

    # File vendor/plugins/gchart/lib/gchart.rb, line 46
46:     def encode(encoding, n, max)
47:       case encoding
48:       when :simple
49:         return "_" if n.nil?
50:         SIMPLE_CHARS[((n/max.to_f) * (SIMPLE_CHARS.size - 1)).round]
51:       when :text
52:         return "-1" if n.nil?
53:         ((((n/max.to_f) * 1000.0).round)/10.0).to_s
54:       when :extended
55:         return "__" if n.nil?
56:         EXTENDED_PAIRS[max.zero? ? 0 : ((n/max.to_f) * (EXTENDED_PAIRS.size - 1)).round]
57:       else
58:         raise ArgumentError, "unsupported encoding: #{encoding.inspect}"
59:       end
60:     end
